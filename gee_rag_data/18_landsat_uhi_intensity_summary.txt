Reference JavaScript snippet (GEE Code Editor):
```javascript
var aoi = ee.Geometry.Rectangle([113.95, 22.28, 114.05, 22.38]); // replace if available
var startDate = '2022-06-01';
var endDate = '2022-09-30';
var minEffectivePixels = 30;

function maskL8L2(image) {
  var qa = image.select('QA_PIXEL');
  var cloudShadowBit = 1 << 4;
  var cloudBit = 1 << 3;
  var clear = qa.bitwiseAnd(cloudShadowBit).eq(0)
    .and(qa.bitwiseAnd(cloudBit).eq(0));
  return image.updateMask(clear);
}

var l8 = ee.ImageCollection('LANDSAT/LC08/C02/T1_L2')
  .filterBounds(aoi)
  .filterDate(startDate, endDate)
  .map(maskL8L2);

var thermal = l8.mean()
  .select('ST_B10')
  .multiply(0.00341802)
  .add(149.0)
  .subtract(273.15)
  .rename('temp_c');

var wc = ee.ImageCollection('ESA/WorldCover/v200').first().select('Map');
var urbanMask = wc.eq(50); // built-up
var ruralMask = wc.neq(50).and(wc.neq(80)); // non-urban and non-water

var urbanPixelCount = thermal.updateMask(urbanMask).reduceRegion({
  reducer: ee.Reducer.count(),
  geometry: aoi,
  scale: 30,
  maxPixels: 1e13
}).get('temp_c');

var ruralPixelCount = thermal.updateMask(ruralMask).reduceRegion({
  reducer: ee.Reducer.count(),
  geometry: aoi,
  scale: 30,
  maxPixels: 1e13
}).get('temp_c');

print('urban_pixel_count', urbanPixelCount);
print('rural_pixel_count', ruralPixelCount);
print('min_effective_pixels', minEffectivePixels);

var hasEnoughUrban = ee.Number(urbanPixelCount).gte(minEffectivePixels);
var hasEnoughRural = ee.Number(ruralPixelCount).gte(minEffectivePixels);
var enoughSamples = hasEnoughUrban.and(hasEnoughRural);

var urbanMean = ee.Algorithms.If(enoughSamples, thermal.updateMask(urbanMask).reduceRegion({
  reducer: ee.Reducer.mean(),
  geometry: aoi,
  scale: 30,
  maxPixels: 1e13
}).get('temp_c'), null);

var ruralMean = ee.Algorithms.If(enoughSamples, thermal.updateMask(ruralMask).reduceRegion({
  reducer: ee.Reducer.mean(),
  geometry: aoi,
  scale: 30,
  maxPixels: 1e13
}).get('temp_c'), null);

var blockedClass = ee.Algorithms.If(hasEnoughUrban, ee.Algorithms.If(hasEnoughRural, null, 'rural'), 'urban');
var uhi = ee.Algorithms.If(
  enoughSamples,
  ee.Number(urbanMean).subtract(ee.Number(ruralMean)),
  null
);

print('blocked_status', ee.Algorithms.If(enoughSamples, 'OK', 'BLOCKED_INSUFFICIENT_SAMPLES'));
print('insufficient_class', blockedClass);
print('urban_mean_temp_c', urbanMean);
print('rural_mean_temp_c', ruralMean);
print('uhi_intensity_c', uhi);
```

Failure modes:
- Ignoring L2 thermal scaling and unit conversion causes invalid temperature values.
- Urban/rural class mapping mismatch leads to unstable contrast estimates.
- Using `ee.Filter.and(...)` / `ee.Filter.or(...)` / `ee.Filter.not(...)` in Python triggers `invalid syntax`.
- AOI resolved from invalid/private boundary asset path causes downstream null geometry errors.
- Running mean/projection on empty image collection can trigger `Image has no bands`.
- Using `FAO/GAUL/2015/level2` as primary boundary source for this Hong Kong district case can miss expected district matches.
- Not inspecting district field names in `projects/ee-hku-geog7310/assets/Hong_Kong_District_Boundary` may cause wrong filter keys and false boundary-missing blocks.
- Returning UHI values when one class has too few pixels creates pseudo-results and must be blocked explicitly.
- Query-intent hijack: forest-loss requests (e.g., "近三年年度森林损失面积 km²") are mistakenly answered by UHI code.
  This is a hard failure; must reroute to Hansen forest-loss profile instead of generating thermal statistics.
- Reporting physically implausible temperatures (e.g., around `-20°C` in Hong Kong summer) due to missing ST_B10 scaling or insufficient QA masking.
