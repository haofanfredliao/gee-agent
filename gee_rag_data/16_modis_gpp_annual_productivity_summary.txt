- Load MOD17A3HGF annual collection and inspect/select productivity band with explicit priority (`Gpp` first).

- Return compact table with fields `year`, `mean_gpp`, `image_count`.

Reference JavaScript snippet (GEE Code Editor):
```javascript
var aoi = ee.Geometry.Rectangle([113.95, 22.28, 114.05, 22.38]); // replace if available
var startYear = 2020;
var endYear = 2022; // fast default window

var annualCol = ee.ImageCollection('MODIS/006/MOD17A3HGF')
  .filterBounds(aoi)
  .filter(ee.Filter.calendarRange(startYear, endYear, 'year'));

var bandNames = ee.List(annualCol.first().bandNames());
var selectedBand = ee.String(ee.Algorithms.If(
  bandNames.contains('Gpp'),
  'Gpp',
  ee.Algorithms.If(bandNames.contains('Npp'), 'Npp', 'NO_VALID_BAND')
));

print('band_names', bandNames);
print('selected_band', selectedBand);

var years = ee.List.sequence(startYear, endYear);

var gppSeries = ee.FeatureCollection(years.map(function(y) {
  y = ee.Number(y).toInt();
  var start = ee.Date.fromYMD(y, 1, 1);
  var end = start.advance(1, 'year');
  var annual = annualCol.filterDate(start, end);
  var imageCount = annual.size();

  var stats = ee.Dictionary(
    ee.Algorithms.If(
      imageCount.gt(0),
      annual.mean().select([selectedBand]).reduceRegion({
        reducer: ee.Reducer.mean(),
        geometry: aoi,
        scale: 500,
        maxPixels: 1e13
      }),
      ee.Dictionary({})
    )
  );

  return ee.Feature(null, {
    year: y,
    image_count: imageCount,
    mean_gpp: ee.Algorithms.If(stats.contains(selectedBand), stats.get(selectedBand), null)
  });
}));

print('modis_gpp_yearly_summary', gppSeries);
```

Expected output:
- Yearly GPP summary table (year, mean_gpp, image_count)

Failure modes:
- Forgetting product scale factor or unit notes causes wrong magnitude.
- Mixing temporal aggregation logic without defining annual boundaries.
- Rewriting GAUL path to `projects/...` aliases can break boundary inspect/load and cascade into null-geometry failures.
- Continuing reductions after both boundary attempts fail (without `BLOCKED_MISSING_BOUNDARY_ASSET`) can produce invalid AOI-based outputs.
- Blocking immediately on private HK boundary failure without trying GAUL fallback causes avoidable task failure.
- Reading reducer output with an unchecked key (for example direct `stats.get('Npp')`) can fail with `Dictionary does not contain key`.
- Using positional band extraction (for example `bandNames().get(0)`) can bind to a non-target band and cause wrong/missing dictionary keys.
- Using `filterDate('2020-01-01', '2022-12-31')` without year-safe handling may miss the intended terminal-year output in some annual products.
